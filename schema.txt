Table Player {
  id integer [primary key]
  name varchar
  age integer
  pace integer
  shooting integer
  passing integer
  dribbling integer
  defending integer
  physicality integer
  diving integer
  handling integer
  kicking integer
  reflexes integer
  positioning integer
}

Table club {
  id integer [primary key]
  name varchar 
  league_id integer
}

Table Contract {
  id integer [primary key]
  player_id integer 
  club_id integer 
  start_date date
  end_date date
  wage integer
  release_clause integer
}

Table Tactic {
  id integer [primary key]
  club_id integer
  formation varchar
  active boolean
  passing_directness integer
  width integer
  tempo integer
  pressing integer
}

Table Match{
  id integer [primary key]
  home_team integer
  away_team integer
  season_id integer
}

Table Result{
  match_id integer [primary key]
  home_goals integer
  away_goals integer
  home_shots integer
  away_shots integer
  home_possession integer
  away_possession integer
  home_corners integer
  away_corners integer
}

Table Signing{
  from_club integer [primary key]
  to_club integer [primary key]
  player_id integer [primary key]
  transfer_date date [primary key]
  transfer_fee integer
  contract_id integer
}

Table League{
  id integer [primary key]
  name varchar 
}

Table Season{
  id integer [primary key]
  year varchar
  league_id integer
}

Table Player_positions {
  player_id integer [primary key]
  position varchar [primary key]
}
